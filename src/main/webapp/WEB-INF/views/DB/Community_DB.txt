
/* 사진 게시판 */
CREATE TABLE PICS (
	PICS_NUM NUMBER PRIMARY KEY, /* 사진 ID */
	PICS_MEMBER NUMBER, /* 글쓴이 */
	PICS_NICK VARCHAR2(50), /* 글쓴이별명 */
	PICS_DATE DATE, /* 작성일자 */
	PICS_CATEGORY VARCHAR2(30), /* 카테고리 */
	PICS_LIKE NUMBER, /* 좋아요 수 */
	PICS_TAG VARCHAR2(100), /* 태그 이름 */
	PICS_READ NUMBER, /* 조회수 */
	PICS_REVIEW NUMBER, /* 후기 */
	PICS_MAIN_IMAGE VARCHAR2(2000), /* 대표사진 */
	PICS_CONTENT VARCHAR2(4000), /* 내용 */
	PICS_PRODUCT NUMBER /* 내용 */
);

CREATE SEQUENCE SEQ_PICS
    START WITH 1
    INCREMENT BY 1;
    
/* 사진 게시판 댓글 */
CREATE TABLE PICS_COMMENT (
	PICS_RE_NUM NUMBER NOT NULL, /* 사진 댓글 ID */
	PICS_RE_PICS NUMBER, /* 달린 사진 번호 */
	PICS_RE_MEMBER NUMBER, /* 글쓴이 */
	PICS_RE_DATE DATE, /* 작성일자 */
	PICS_RE_CONTENT VARCHAR2(500), /* 내용 */
	PICS_RE_REF NUMBER, /* 원글답글 묶어 */
	PICS_RE_STEP NUMBER, /* 답글 순서 정의 */
	PICS_RE_LEVEL NUMBER /* 들여쓰기 효과 */
);
ALTER TABLE PICS_COMMENT
	ADD
		CONSTRAINT PK_PICS_COMMENT
		PRIMARY KEY (
			PICS_RE_NUM
		);

CREATE SEQUENCE SEQ_MEMBER_LIKE
    START WITH 1
    INCREMENT BY 1;

/* 팔로우 */
CREATE TABLE FOLLOW (
	FOLLOW_NUM NUMBER PRIMARY KEY, /* 팔로우 ID */
	FOLLOW_FROM NUMBER, /* 회원 번호 */
	FOLLOW_TO NUMBER /* 팔로우 대상 */
);
CREATE SEQUENCE SEQ_FOLLOW
    START WITH 1
    INCREMENT BY 1;
